FAIL - First Atari Image Library
================================

link:/[Home]

This page decodes pictures in formats native to Atari computers.
Decoding takes place in your browser, nothing is sent to a server.
You can http://caniuse.com/filereader[check] if your browser is modern enough.

<script type="text/javascript" src="fail.js"> </script>
<script type="text/javascript" src="html5fail.js"> </script>
<script type="text/javascript">
function openFile(file)
{
	var reader = new FileReader();
	reader.onload = function (e) {
		fail2canvas(file.name, new Uint8Array(e.target.result));
	};
	reader.readAsArrayBuffer(file);
}

function dragHelper(e)
{
	e.stopPropagation();
	e.preventDefault();
}

function onDrop(e)
{
	dragHelper(e);
	openFile(e.dataTransfer.files[0]);
}
</script>

Select file: <input type="file" onchange="openFile(this.files[0])" />
<div style="border: solid #800080 1px; padding-top: 2em; padding-bottom: 2em;" ondragenter="dragHelper(event)" ondragover="dragHelper(event)" ondrop="onDrop(event)">
